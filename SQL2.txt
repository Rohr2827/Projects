DROP TABLE SUN_CUSTOMER CASCADE CONSTRAINTS;
DROP TABLE SUN_VENDOR CASCADE CONSTRAINTS;
DROP TABLE SUN_RAW_MATERIAL CASCADE CONSTRAINTS;
 

CREATE TABLE SUN_CUSTOMER
(
	CUST_CODE		NUMBER(10)			,	
	CUST_BUSINESS_NAME	VARCHAR2(40)	NOT NULL	,
	CUST_AREA_CODE		NUMBER(3)	NOT NULL	,
	CUST_PHONE_NUM		NUMBER(7)	NOT NULL	,
	CONSTRAINT SUN_CSTMR_CST_CD_PK PRIMARY KEY (CUST_CODE)
);

CREATE TABLE SUN_VENDOR
(
	VENDOR_CODE		NUMBER(10)			,
	VENDOR_BUSINESS_NAME	VARCHAR2(40)	NOT NULL	,
	VENDOR_AREA_CODE	NUMBER(3)	NOT NULL	,
	VENDOR_PHONE_NUM	NUMBER(7)	NOT NULL	,
	CONSTRAINT SUN_VNDR_VND_CD_PK PRIMARY KEY (VENDOR_CODE)
);

CREATE TABLE SUN_RAW_MATERIAL
(
	RAW_MATERIAL_CODE	NUMBER(10)			,
	RAW_MAT_NAME		VARCHAR(40)	NOT NULL	,
	RAW_MAT_QTY_PER_UNIT 	NUMBER(10)	 		,
	RAW_MAT_UNIT		VARCHAR2(35)			,
	RAW_MAT_PRICE_PER_UNIT	NUMBER(12,2)    NOT NULL        , 
	RAW_MAT_STRG_INSTRUCT	VARCHAR2(75)	 		,
	CONSTRAINT RAW_MTRL_MTRL_CD_PK PRIMARY KEY (RAW_MATERIAL_CODE) 
);

CREATE TABLE SUN_RECIPE
(
	RECIPE_CODE		NUMBER(10)		 	,
	RECIPE_TITLE		VARCHAR2(40)			,	
	RECIPE_DESC		VARCHAR2(75) 	  		,
	RECIPE_SPEC_INSTRUCT	VARCHAR2(75)		  	,
	CONSTRAINT SUN_RECIPE_RCPE_CD_PK PRIMARY KEY (RECIPE_CODE)
);


CREATE TABLE SUN_PASTRY
(
	PASTRY_CODE		NUMBER(10)			,
	PASTRY_NAME		VARCHAR2(40)	NOT NULL	,			
	PASTRY_DESC		VARCHAR2(75) 	       		,
	PASTRY_PRICE_PER_UNIT	NUMBER(12,2)    NOT NULL        , 
	RECIPE_CODE		NUMBER(10)	NOT NULL	,
	PASTRY_BAKED_DATE	DATE		NOT NULL	,
	CONSTRAINT SUN_PASTRY_PSTRY_CD_PK PRIMARY KEY (PASTRY_CODE),
	CONSTRAINT SUN_PASTRY_RECPE_CD_FK FOREIGN KEY (RECIPE_CODE) REFERENCES SUN_RECIPE(RECIPE_CODE)
);

CREATE TABLE SUN_ORDER
(
	ORDER_NUM		NUMBER(10)			,	
	ORDER_DATE		DATE		NOT NULL	,
	CUST_CODE		NUMBER(10)	NOT NULL	,
	CONSTRAINT SUN_ORDR_ORDR_NUM_PK PRIMARY KEY (ORDER_NUM)	, 
	CONSTRAINT SUN_ORDR_CUST_CODE_FK FOREIGN KEY (CUST_CODE) REFERENCES SUN_CUSTOMER(CUST_CODE)
);

CREATE TABLE SUN_ORDER_LINE
(
	ORDER_NUM		NUMBER(10)			,
	ORDER_LINE_NUM		NUMBER(10)	 		,
	PASTRY_CODE		NUMBER(10)	NOT NULL	,
	QUANTITY_ORDERED	NUMBER(10)	NOT NULL    	,
	CONSTRAINT SUN_ORDER_NUM_LINE_PK PRIMARY KEY (ORDER_NUM,ORDER_LINE_NUM),
	CONSTRAINT SUN_ORDR_NUM_FK FOREIGN KEY (ORDER_NUM) REFERENCES SUN_ORDER (ORDER_NUM), 
	CONSTRAINT SUN_ORDR_LNE_PASTRY_CD_FK FOREIGN KEY (PASTRY_CODE) REFERENCES SUN_PASTRY (PASTRY_CODE) 
);

CREATE TABLE SUN_RECIPE_LINE
(
	RECIPE_CODE		NUMBER(10)	 		,
	RECIPE_LINE_NUM		NUMBER(10)	 		,
	RAW_MATERIAL_CODE	NUMBER(10)	NOT NULL	,
	RAW_MAT_AMOUNT		NUMBER(10)	NOT NULL	,
	RAW_MAT_AMT_UNIT	VARCHAR2(25)			,
	RECIPE_INGRED_SPEC_INST	VARCHAR2(75)			,
	CONSTRAINT SUN_RECIPE_RCPE_LINE_PK PRIMARY KEY (RECIPE_CODE,RECIPE_LINE_NUM),
	CONSTRAINT SUN_RECIPE_RW_MTRL_CD_FK FOREIGN KEY (RAW_MATERIAL_CODE) REFERENCES SUN_RAW_MATERIAL(RAW_MATERIAL_CODE)	
);

CREATE TABLE SUN_VEND_TO_RAW_MAT
(
	VENDOR_CODE		NUMBER(10)	NOT NULL,
	RAW_MATERIAL_CODE	NUMBER(10)	NOT NULL,
	CONSTRAINT SUN_VEND_TO_RAW_MAT_PK  PRIMARY KEY (VENDOR_CODE,RAW_MATERIAL_CODE),
	CONSTRAINT SUN_VEND_TO_RAW_MAT_FK1 FOREIGN KEY (VENDOR_CODE) REFERENCES SUN_VENDOR (VENDOR_CODE),
	CONSTRAINT SUN_VEND_TO_RAW_MAT_FK2 FOREIGN KEY (RAW_MATERIAL_CODE) REFERENCES SUN_RAW_MATERIAL (RAW_MATERIAL_CODE)	
);
